  {

    {
      Thread thread;
      {
        thread = Thread.currentThread();
      }
      public void run()
      {
        while( thread.isAlive() )
        {
              FileTracer.handle( System.currentTimeMillis()+","+Thread.currentThread().getId()+",TraceSingleAttributePeriod.ump,8,Tracer,"+System.identityHashCode(this)+",at_p,id,"+id );
          try
          {
            Thread.sleep(100s);
          }

            e.printStackTrace();
          }
        }

    }
    ).start();
